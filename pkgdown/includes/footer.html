document.addEventListener('DOMContentLoaded', function () { var page = window.location.pathname; var footerFile = ''; if (page.includes('/Start')) { footerFile = 'footer_Start.html'; } else if (page.includes('/news')) { footerFile = 'footer_news.html'; } if (footerFile) { fetch(`pkgdown/includes/${footerFile}`) .then(response => response.text()) .then(data => { document.querySelector('.page-footer').innerHTML = data; }) .catch(error => { console.error('Error loading footer:', error); }); } });
